# PROCEDURA CON OVS - connessione della macchina allo switch virtuale
sudo ovs-vsctl --if-exists del-br gateway1
sudo dhclient enp0s8
sudo ovs-vsctl add-br gateway1
sudo ovs-vsctl add-port gateway1 enp0s8
sudo ip addr flush dev enp0s8
sudo dhclient gateway1

-> Prende l'indirizzo che solitamente si prende enp0s8
A questo punto la macchina è connessa

# Creazione di una veth, simula un cavo connesso ad uno switch
# veth1 è l'estremità connessa allo switch, veth1-peer è l'estremità connessa al Router Arista
sudo ip link add veth1 type veth peer name veth1-peer -> Da qui in poi no
sudo ip link set veth1 up
sudo ip link set veth1-peer up
sudo ovs-vsctl add-port gateway1 veth1

-> La macchina deve usare veth1-peer

ovs-vsctl show -> Verificare la connessione
